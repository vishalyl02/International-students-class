







// Question 1: Addition of Integers
#include <stdio.h>
int main() {
    int a = 10, b = 20;
    printf("%d\n", a + b);
    return 0;
}




// Question 2: Subtraction of Floats
#include <stdio.h>
int main() {
    float x = 5.5, y = 2.2;
    printf("%.2f\n", x - y);
    return 0;
}





// Question 3: Multiplication of Integers
#include <stdio.h>
int main() {
    int p = 7, q = 6;
    printf("%d\n", p * q);
    return 0;
}




// Question 4: Division with Float Result
#include <stdio.h>
int main() {
    float num = 10.0, denom = 4.0;
    printf("%.2f\n", num / denom);
    return 0;
}





// Question 5: Combining Operations
#include <stdio.h>
int main() {
    float a = 8.5, b = 3.0;
    printf("%.2f\n", (a + b) * b);
    return 0;
}
